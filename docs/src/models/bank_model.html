 61        # Read interest data

 67        self.zerocurve = Zerocurve()

 69        # read zerocurve data
 70        self.zerocurve.read_data()
 71        start_date, end_date = self.zerocurve.get_period()
 72        self.interest.set_period(start_date, end_date)

 76
 77        # fit the Hull-White model to simulate interest rates
 78        self.hullwhite = HullWhiteModel()
 79        self.hullwhite.fit(self.interest, self.zerocurve)
 80        self.hullwhite.transform()

 82        # initialize the remaining fields for the bank model
 83        self.pos_date = end_date.to_numpy()
 84        self.timestep = 0

 92    def _random_date_(self, start: datetime, end: datetime) -> datetime:
 93        """return a random business date between start and end date"""
 94        delta = end - start
 95        date = start + timedelta(days=randrange(delta.days))
 96        # shift dates to next business day if closed
 97        if date.weekday() >= 5:
 98            date = date + BDay(1)
 99            date = date.to_pydatetime()
100        return date

102    def generate_mortgage_contracts(
103        self,
104        n: int = MORTGAGE_SIZE,
105        amount: float = MORTGAGE_AMOUNT,
106        probabilities=RANGE_PROBABILITIES,
107    ):
108        """Generate mortgage contracts for a specific period"""
109
110        """ If we can't fund the mortgages - we can not sell more mortgages """
111        if self.liquidity < n * amount:
112            # logger.debug(
113            #     f"Not enough liquidity to fund mortgages. Mortgage size {n}, amount {amount}, {self.liquidity} < {amount*n}"
114            # )
115            return
116
117        # We use structured arrays to store the data - as dataframes are too slow
118        dtypes = COLUMN_DATA
119        data = np.empty(n, dtype=dtypes)
120        data["tenor"] = np.random.choice(RANGE_TENOR, size=n, p=probabilities)
121        data["start_date"] = np.full(n, self.pos_date)
122        delta = np.timedelta64(365, "D") * data["tenor"]
123        data["maturity_date"] = data["start_date"] + delta
124        data["principal"] = np.ones(n) * amount
125        data["period"] = np.array([np.datetime64(dt, "Y") for dt in data["start_date"]])
126        data["interest"] = np.array(
127            [
128                self.get_sim_interest_rate(self.timestep, tenor)
129                for tenor in data["tenor"]
130            ]
131        )
132        self.liquidity -= n * amount
133        if self.sa_mortgages is None:
134            self.sa_mortgages = data
135        else:
136            self.sa_mortgages = np.concatenate((self.sa_mortgages, data))

138    def get_sim_interest_rate(self, timestep, tenor):
139        """Get interest rate for a specific start date and tenor"""
140        x = RANGE_TENOR.index(tenor)
141        rate = self.hullwhite.get_simulated_interest_rates(timestep)[x, 0]
142        return rate

144    def get_sim_zero_rate(self, timestep, tenor):
145        """Get zero rate for a specific start date and tenor"""
146        rates = self.hullwhite.get_simulated_zero_rates(timestep)[:, 0]
147        tenor_per_month = tenor * 12
148
149        TENORS = {
150            "ON": 0,
151            "IF_3M": 3,
152            "IF_6M": 6,
153            "IF_9M": 9,
154            "IF_1Y": 12,
155            "IF_1Y3M": 15,
156            "IF_1Y6M": 18,
157            "IF_2Y": 24,
158            "IF_3Y": 36,
159            "IF_4Y": 48,
160            "IF_5Y": 60,
161            "IF_7Y": 84,
162            "IF_10Y": 120,
163            "IF_15Y": 180,
164            "IF_30Y": 360,
165        }
166
167        # Extract the tenor values from the TENORS dictionary
168        tenor_values = np.array(list(TENORS.values()))
169
170        # Linearly interpolate the rate using numpy's interp function
171        interpolated_rate = np.interp(tenor_per_month, tenor_values, rates)
172
173        return interpolated_rate

202    def step(self, actions=None, timestep=None) -> None:
203        """Each step we fund a time bucket or not.
204        We asume we can not repay the funding before maturity."""
205        if timestep is not None:
206            self.timestep = timestep
207        from_date = self.pos_date
208        self.pos_date = np.datetime64(
209            self.pos_date.astype("datetime64[M]") + np.timedelta64(1, "M")
210        )
211        if actions is not None:
212            tenor = 0
213            for action in actions:
214                if action == 0:
215                    pass
216                else:
217                    self.buy_sell_bond(buy_sell=action, tenor=FUNDING_TENORS[tenor])
218                tenor += 1
219
220        # receive payments from mortgages
221        # TODO: check liquidity calculation. It may not be correct
222        if self.sa_mortgages is not None:
223            self.liquidity += np.where(
224                (self.sa_mortgages["maturity_date"] >= from_date)
225                & (self.sa_mortgages["maturity_date"] < self.pos_date),
226                self.sa_mortgages["principal"],
227                0,
228            ).sum()
229        # pay back funding
230        if self.sa_funding is not None:
231            self.liquidity += np.where(
232                (self.sa_funding["maturity_date"] >= from_date)
233                & (self.sa_funding["maturity_date"] < self.pos_date),
234                self.sa_funding["principal"],
235                0,
236            ).sum()
237
238        self.generate_mortgage_contracts(int((MORTGAGE_SIZE + randint(-1, 1)) / 12))
239        self.timestep = self.timestep + 1

241    def calculate_nii(self) -> Tuple[float, float, float]:
242        """Calculate the Net Interest Income per period"""
243
244        income = 0
245        funding_cost = 0
246
247        if self.sa_mortgages is not None:
248            filter = self.sa_mortgages["maturity_date"] >= self.pos_date
249            mortgages = self.sa_mortgages[filter]
250            income = ((mortgages["interest"] / 100) * mortgages["principal"] / 12).sum()
251
252        if self.sa_funding is not None:
253            filter = self.sa_funding["maturity_date"] >= self.pos_date
254            funding = self.sa_funding[filter]
255            funding_cost = (
256                (funding["interest"] / 100) * funding["principal"] / 12
257            ).sum()
258
259        nii = income + funding_cost
260        return nii, income, funding_cost

262    def calculate_cashflows(self, type: str = "all") -> np.ndarray:
263        """Calculate the future principal cashflows from mortgages and funding"""
264
265        # Create cashflow_data array
266        projected_years = 31  # added 1 for the current year
267        pos_date = np.datetime64(self.pos_date)
268        end_date = np.datetime64(
269            pos_date.astype("datetime64[Y]") + np.timedelta64(projected_years, "Y")
270        )
271        dtype = [("year", "datetime64[Y]"), ("cashflow", int)]
272        cashflow_data = np.zeros(projected_years, dtype=dtype)
273        cashflow_data["year"] = np.arange(pos_date, end_date, dtype="datetime64[Y]")
274
275        if type not in ["mortgages", "funding", "all"]:
276            raise ValueError("type must be mortgages, funding or all")
277        if type == "mortgages":
278            if self.sa_mortgages is None:
279                return cashflow_data
280            data = self.sa_mortgages
281        elif type == "funding":
282            if self.sa_funding is None:
283                return cashflow_data
284            data = self.sa_funding
285        else:
286            if self.sa_funding is None:
287                if self.sa_mortgages is None:
288                    return cashflow_data
289                data = self.sa_mortgages
290            else:
291                if self.sa_mortgages is None:
292                    data = self.sa_funding
293                else:
294                    data = np.concatenate((self.sa_funding, self.sa_mortgages))
295
296        maturity_dates = np.array(data["maturity_date"], dtype="datetime64[Y]")
297        principals = np.array(data["principal"], dtype=float)
298        start_dates = np.array(data["start_date"], dtype="datetime64[Y]")
299
300        # Calculate cashflows
301
302        # receivables
303        rec_mask = maturity_dates >= pos_date
304        rec_value_date = maturity_dates[rec_mask]
305        rec_cashflows = principals[rec_mask]
306        rec_indices = np.searchsorted(
307            cashflow_data["year"],
308            rec_value_date.astype("datetime64[Y]"),
309        )
310        np.add.at(cashflow_data["cashflow"], rec_indices, rec_cashflows)
311
312        # payments
313        pay_mask = start_dates >= pos_date
314        pay_value_date = start_dates[pay_mask]
315        pay_cashflows = -1 * principals[pay_mask]
316        pay_indices = np.searchsorted(
317            cashflow_data["year"],
318            pay_value_date.astype("datetime64[Y]"),
319        )
320        np.add.at(cashflow_data["cashflow"], pay_indices, pay_cashflows)
321
322        return cashflow_data

324    def reset(self):
325        self.timestep = 0
326        self.liquidity = 0  # We don't have any cash at the beginning
327        _, end_date = self.interest.get_period()
328        self.pos_date = end_date.to_numpy()
329        self.hullwhite.transform()  # Start with new simulated rates
330        self.sa_mortgages = None
331        self.sa_funding = None

333    def get_risk_penalty(self):
334        cf = self.calculate_cashflows("all")
335        risk_penalty = (
336            (
337                abs(
338                    cf[abs(cf["cashflow"]) > self.risk_limit]["cashflow"]
339                    - self.risk_limit
340                ).sum()
341                / self.risk_limit
342            )
343            * MORTGAGE_AMOUNT
344            / 12
345        )
346        return risk_penalty
347
348    def get_reward(self):
349        nii, income, funding_cost = self.calculate_nii()
350        cf = self.calculate_cashflows("all")
351
352        # Liquidity penalty if we can not pay the projected cashflows
353        if cf[0]["cashflow"] + self.liquidity < 0:
354            # liquidity_penalty = MORTGAGE_AMOUNT / 12
355            liquidity_penalty = 0
356        else:
357            liquidity_penalty = 0
358        risk_penalty = self.get_risk_penalty()
359        reward = nii - risk_penalty - liquidity_penalty
360        return int(reward), int(nii), int(risk_penalty), int(liquidity_penalty)
361
362    def draw_cashflows(self):
363        c = self.calculate_cashflows()
364        plt.title("Expected Cashflows")
365        plt.xlabel("X axis")
366        plt.ylabel("Y axis")
367        x = np.arange(
368            self.pos_date,
369            self.pos_date + np.timedelta64(31, "Y"),
370            dtype="datetime64[Y]",
371        )
372        y = c["cashflow"]
373        plt.plot(x, y, "-")
374        plt.show()

377def main():
378    bankmodel = Bankmodel()
379    bankmodel.reset()
380    actionspace = ActionSpace(len(FUNDING_TENORS), 0, BOND_AMOUNT)
381
382    score = 0
383    for _ in range(60):
384        bankmodel.step(actionspace.actionspace.sample())
385        reward, _, _, _ = bankmodel.get_reward()
386        score = score + reward
387    print(f"score = {score}")
388    bankmodel.draw_cashflows()
